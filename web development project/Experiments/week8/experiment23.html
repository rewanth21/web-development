First Experiment with SVG

Revolving triangle indefinitely over a faded circle. This is my first experiment with SVG using codepen as reference

  <radialGradient id="baseGradient">
			<stop offset="050%" stop-color="#ff5" />
			<stop offset="100%" stop-color="#ccc" />
		</radialGradient>
	</defs>

	<circle id="base" cx="50" cy="50" r="15" fill="url(#baseGradient)" stroke-width="1" stroke="#aaa">
            <animate attributeType="xml" attributeName="r" from="0" to="25" dur="1s" />
	</circle>

	<path d="m50,15 l5,10 l-10,0 l5,-10 z" fill="#eee3" stroke="#aaa">
		<animateTransform attributeName="transform" type="rotate" from="0 50 50" to="360 50 50" dur=".5s" repeatCount="indefinite" />
		<animateTransform attributeName="transform" type="scale" from="0" to="1" dur=".5s" />
	</path>
